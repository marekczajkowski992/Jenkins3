pipeline {
    agent { 
        label 'main'
    }
    tools {
        maven 'Maven_3_8_4'
    }
    stages {
        stage('Checkout') {
            steps {
                checkout scm // Checkout code from SCM
            }
        }
        stage('Build') {
            steps {
                sh 'mvn clean install -DskipTests'
                sh 'mvn compile'
            }
        }
        stage('Test') {
            steps {
                sh 'mvn clean test'
            }
        }
    }
}
